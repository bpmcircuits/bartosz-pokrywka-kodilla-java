SELECT IL.NAME, I.SUMMARY, I.DESCRIPTION
FROM ISSUES I JOIN ISSUESLISTS IL
ORDER BY IL.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_COUNT
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME
ORDER BY ISSUES_COUNT;
